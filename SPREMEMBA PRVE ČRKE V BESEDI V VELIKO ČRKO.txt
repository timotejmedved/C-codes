#include <stdio.h>      
#include <stdlib.h> 
#include <iostream>
#include <time.h> 
#include <cstdlib>

using namespace std;


int main(int argc, char** argv) {

    char besedilo[80];
    
    cout<<"VNESITE BESEDILO. "<<endl;
    cin.getline(besedilo,80);
    
    //spremeni velike črke v male
    
    for(int i=0;i<80;i++)
    {
        if('A'<=besedilo[i] && besedilo[i]<='Z')
        {
            besedilo[i]=char(((int)besedilo[i])+32);
        }
            
    }
    
   //sprememba PRVE črke v besedi v veliko črko

    int i=0;

    for(int i=0; i<80; i++){

        if(i==0){besedilo[i]=char(((int)besedilo[i])-32);}
        
        if(besedilo[i]==' ' ){
            i++;
           besedilo[i]=char(((int)besedilo[i])-32);
           i--; 
        
        }
        
    }
    
    
    
    cout<<besedilo<<"";
    
    return 0;
}
//konec programa sprememba PRVE črke v veliko črko
